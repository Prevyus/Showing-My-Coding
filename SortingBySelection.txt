#include <stdio.h>

int Arreglo[5] ={25, 13, 31, 17, 2 } ;
void Ordena_Seleccion();

void main()
{
    int i = 0;
    printf ( "\n\n ORDENAMIENTO POR SELECCION \n");
    printf ("\n\n Arreglo antes el ordenamiento:\n");

    for(i=0; i<=4; i++)
    {
        printf ("%d\t", Arreglo[i] );
    }
       
        Ordena_Seleccion();

    printf ("\n\nArreglo despues el ordenamiento:\n");

    for(i=0; i<=4; i++)
    {
        printf ("%d\t", Arreglo[i] ) ;
    }
}

void Ordena_Seleccion()
{
    int i=0, j=0, temp=0;
   
    for (i=0;i<=3; i++)
    { 
        for (j=i+1;j<=4 ; j++ )
        {
            if ( Arreglo[i] > Arreglo[j] )
            {
            temp = Arreglo[i] ;
            Arreglo[i] = Arreglo[j];
            Arreglo[j] = temp ;
            }
        }
    }
}
